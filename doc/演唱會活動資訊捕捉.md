### 抓取邏輯.ts

```
export default {
  async fetch(request: Request, env: any): Promise<Response> {
    const url = new URL(request.url);
    const startAt = url.searchParams.get("start_at") || "";
    const endAt = url.searchParams.get("end_at") || "";

    let allEvents: any[] = [];
    let page = 1;
    const MAX_PAGES = 5; // 安全限制：最多連續抓 5 頁，避免 CPU 逾時

    try {
      while (page <= MAX_PAGES) {
        // 1. 構建該頁的 URL
        const kktixUrl = `https://kktix.com/events?utf8=✓&event_tag_ids_in=1&start_at=${encodeURIComponent(startAt)}&end_at=${encodeURIComponent(endAt)}&page=${page}`;

        const response = await fetch(kktixUrl, {
          headers: { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36..." }
        });
        const html = await response.text();

        // 2. 解析當前頁面的活動
        const regex = /<li class="event-listing-item">[\s\S]*?<h2><a href="([^"]+)">([^<]+)<\/a><\/h2>/g;
        let match;
        let countInPage = 0;

        while ((match = regex.exec(html)) !== null) {
          allEvents.push({
            url: match[1],
            title: match[2].trim(),
            id: match[1].split('/').pop(),
            foundInPage: page
          });
          countInPage++;
        }

        // 3. 判斷是否需要抓下一頁
        // 如果這一頁沒抓到東西，或者抓到的數量少於 KKTIX 的常規數量 (20)，代表沒下一頁了
        if (countInPage === 0 || countInPage < 20) {
          break;
        }

        page++;
        // 稍微等待 (Optional)，避免被 KKTIX 偵測
        // await new Promise(resolve => setTimeout(resolve, 200)); 
      }

      return new Response(JSON.stringify({ 
        success: true, 
        total: allEvents.length,
        pagesFetched: page,
        data: allEvents 
      }), {
        headers: { "Content-Type": "application/json;charset=UTF-8", "Access-Control-Allow-Origin": "*" }
      });

    } catch (error) {
      return new Response(JSON.stringify({ success: false, error: error.message }), { status: 500 });
    }
  }
};
```

```
    const startStr = today.toISOString().split('T')[0].replace(/-/g, '/');
    const endStr = nextMonth.toISOString().split('T')[0].replace(/-/g, '/');

    try {
      const res = await fetch(`https://your-worker.workers.dev?start_at=${startStr}&end_at=${endStr}`);
      const json = await res.json();
      if (json.success) {
        setData({ events: json.data, stats: { total: json.total, pages: json.pagesFetched } });
      }
    } catch (e) {
      alert("抓取失敗");

      setLoading(false);

  );
};
```
